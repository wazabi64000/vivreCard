4 — Serveur Express


// server.js
import express from "express";
import cors from "cors";
import { db } from "./db.js";

const app = express();

app.use(cors());            // Autoriser le frontend
app.use(express.json());     // Lire le JSON

// Endpoint POST : créer un utilisateur
app.post("/api/users", async (req, res) => {
  const { name, email } = req.body;
  await db.execute(
    "INSERT INTO users (name, email) VALUES (?, ?)",
    [name, email]
  );
  res.status(201).json({ message: "Utilisateur enregistré" });
});

// Endpoint GET : récupérer tous les utilisateurs
app.get("/api/users", async (req, res) => {
  const [rows] = await db.execute("SELECT * FROM users");
  res.json(rows);
});

// Démarrage serveur
app.listen(5000, () => {
  console.log("Serveur sur http://localhost:5000");
});

5 — Frontend React + Axios + dossier API


// /src/api/userApi.js
import axios from "axios";

// Crée une instance Axios pour réutiliser la base URL
const API = axios.create({
  baseURL: "http://localhost:5000/api"
});

// Fonction pour créer un utilisateur
export const createUser = (data) => {
  return API.post("/users", data);
};

// Fonction pour récupérer les utilisateurs
export const getUsers = () => {
  return API.get("/users");
};

6 — Formulaire React avec React Hook Form


// App.jsx
import { useForm } from "react-hook-form";
import { createUser, getUsers } from "./api/userApi";
import { useEffect, useState } from "react";

function App() {
  const { register, handleSubmit, reset } = useForm();
  const [users, setUsers] = useState([]);

  const onSubmit = async (data) => {
    await createUser(data);
    loadUsers();
    reset();
  };

  const loadUsers = async () => {
    const res = await getUsers();
    setUsers(res.data);
  };

  useEffect(() => {
    loadUsers();
  }, []);

  return (
    <div>
      <h2>Créer un utilisateur</h2>
      <form onSubmit={handleSubmit(onSubmit)}>
        <input placeholder="Nom" {...register("name")} />
        <input placeholder="Email" {...register("email")} />
        <button type="submit">Envoyer</button>
      </form>

      <h3>Liste utilisateurs</h3>
      {users.length > 0 && (
        <ul>
          {users.map(user => (
            <li key={user.id}>
              {user.name} - {user.email}
            </li>
          ))}
        </ul>
      )}
      {users.length === 0 && <p>Aucun utilisateur trouvé</p>}
    </div>
  );
}

export default App;
